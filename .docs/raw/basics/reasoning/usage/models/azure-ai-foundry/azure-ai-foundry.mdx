title: Azure AI Foundry
---

<Steps>

  <Step title="Create a Python file">
    ```bash
    touch azure_ai_foundry.py
    ```
  </Step>


  <Step title="Add the following code to your Python file">
    ```python azure_ai_foundry.py
    import os

    from agno.agent import Agent
    from agno.models.azure import AzureAIFoundry

    agent = Agent(
        model=AzureAIFoundry(id="gpt-5-mini"),
        reasoning_model=AzureAIFoundry(
            id="DeepSeek-R1",
            azure_endpoint=os.getenv("AZURE_ENDPOINT"),
            api_key=os.getenv("AZURE_API_KEY"),
        ),
    )

    agent.print_response(
        "Solve the trolley problem. Evaluate multiple ethical frameworks. "
        "Include an ASCII diagram of your solution.",
        stream=True,
    )
    ```
  </Step>

  <Snippet file="create-venv-step.mdx" />

  <Step title="Install libraries">
    ```bash
    pip install -U azure-ai-inference agno
    ```
  </Step>

  <Step title="Set your Azure API credentials">

    <CodeGroup>

    ```bash Mac/Linux
      export AZURE_API_KEY="xxx"
      export AZURE_ENDPOINT="xxx"
    ```

    ```bash Windows
      $Env:AZURE_API_KEY="xxx"
      $Env:AZURE_ENDPOINT="xxx"
    ```
    </CodeGroup>
  </Step>
  <Step title="Run Agent">
    <CodeGroup>
    ```bash Mac
    python azure_ai_foundry.py
    ```

    ```bash Windows
    python azure_ai_foundry.py
    ```
    </CodeGroup>
  </Step>

  
</Steps>
