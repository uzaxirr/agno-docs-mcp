title: Azure OpenAI GPT 4.1

<Steps>

  <Step title="Create a Python file">
    ```bash
    touch azure_openai.py
    ```
  </Step>


  <Step title="Add the following code to your Python file">
    ```python azure_openai.py
    from agno.agent import Agent
    from agno.models.azure.openai_chat import AzureOpenAI

    agent = Agent(
        model=AzureOpenAI(id="gpt-5-mini"), 
        reasoning_model=AzureOpenAI(id="gpt-4.1")
    )
    agent.print_response(
        "Solve the trolley problem. Evaluate multiple ethical frameworks. "
        "Include an ASCII diagram of your solution.",
        stream=True,
    )
    ```
  </Step>

  <Snippet file="create-venv-step.mdx" />

  <Step title="Install libraries">
    ```bash
    pip install -U openai agno
    ```
  </Step>

  <Step title="Set your Azure OpenAI credentials">

    <CodeGroup>

    ```bash Mac/Linux
      export AZURE_OPENAI_API_KEY="xxx"
      export AZURE_OPENAI_ENDPOINT="xxx"
      export AZURE_DEPLOYMENT="xxx"
    ```

    ```bash Windows
      $Env:AZURE_OPENAI_API_KEY="xxx"
      $Env:AZURE_OPENAI_ENDPOINT="xxx"
      $Env:AZURE_DEPLOYMENT="xxx"
    ```
    </CodeGroup>
  </Step>
  <Step title="Run Agent">
    <CodeGroup>
    ```bash Mac
    python azure_openai.py
    ```

    ```bash Windows
    python azure_openai.py
    ```
    </CodeGroup>
  </Step>

</Steps>
